givenphp

开局直接给了个代码：

<?php
highlight_file(__FILE__);  // 将当前文件的源代码进行语法高亮并输出

// 处理文件上传
if(isset($_POST['upload'])){  // 检查是否存在名为 'upload' 的 POST 请求参数
    handleFileUpload($_FILES['file']);  // 调用 handleFileUpload() 函数处理文件上传
}

// 处理 GET 请求参数
if(isset($_GET['challenge'])){  // 检查是否存在名为 'challenge' 的 GET 请求参数
    waf();  // 调用 waf() 函数进行输入过滤和防御
    $value=$_GET['value'];  // 获取名为 'value' 的 GET 请求参数的值
    $key=$_GET['key'];  // 获取名为 'key' 的 GET 请求参数的值
    
    // 使用 create_function() 创建一个匿名函数，并将其赋值给 $func
    $func=create_function("","putenv('$key=$value');");
    
    // 检查 $func 是否等于名为 'guess' 的 GET 请求参数的值
    if($func==$_GET['guess']){
        $func();  // 调用 $func() 函数
        system("whoami");  // 执行系统命令 "whoami"
    }
}

// 输入过滤和防御函数
function waf()
{
    // 使用正则表达式检查 GET 请求参数 'key' 和 'value' 中是否包含恶意字符串
    if(preg_match('/\'|"|%|\(|\)|;|bash/i',$_GET['key'])||preg_match('/\'|"|%|\(|\)|;|bash/i',$_GET['value'])){
        die("evil input!!!");  // 如果包含恶意字符串，输出错误信息并终止脚本的执行
    }
}

// 处理文件上传的函数
function handleFileUpload($file)
{
    $uploadDirectory = '/tmp/';  // 上传目录的路径

    if ($file['error'] !== UPLOAD_ERR_OK) {  // 检查文件上传是否成功
        echo '文件上传失败。';  // 如果上传失败，输出错误信息
        return;
    }

    $fileExtension = pathinfo($file['name'], PATHINFO_EXTENSION);  // 获取上传文件的扩展名

    $newFileName = uniqid('uploaded_file_', true) . '.' . $fileExtension;  // 生成新的文件名
    $destination = $uploadDirectory . $newFileName;  // 设置文件的目标路径

    if (move_uploaded_file($file['tmp_name'], $destination)) {  // 将上传的文件移动到目标路径
        echo $destination;  // 输出移动后的文件路径
    } else {
        echo '文件移动失败。';  // 如果文件移动失败，输出错误信息
    }
}
