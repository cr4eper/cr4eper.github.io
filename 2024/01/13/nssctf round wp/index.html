<?php
error_reporting(0);
highlight_file(__FILE__);

// 包含名为 "rce.php" 的外部文件
include("rce.php");

// 初始化四个检查器变量
$checker_1 = FALSE;
$checker_2 = FALSE;
$checker_3 = FALSE;
$checker_4 = FALSE;

// 从 $_GET 超全局变量中获取 "num" 参数的值
$num = $_GET['num'];

// 检查 "num" 参数是否包含任何数字
if (preg_match("/[0-9]/", $num)) {
    die("no!!");
}

// 检查 "num" 参数是否可以转换为整数
if (intval($num)) {
    $checker_1 = TRUE;
}

// 检查 $_POST 超全局变量中是否同时设置了 "ctype" 和 "is_num" 参数
if (isset($_POST['ctype']) && isset($_POST['is_num'])) {
    $ctype = strrev($_POST['ctype']);
    $is_num = strrev($_POST['is_num']);
    
    // 检查 "ctype" 是否只包含字母字符，
    // "is_num" 是否为数字，以及它们的 MD5 哈希是否相等
    if (ctype_alpha($ctype) && is_numeric($is_num) && md5($ctype) == md5($is_num)) {
        $checker_2 = TRUE;
    }
}

// 从 $_GET 和 $_POST 超全局变量中获取 "114" 和 "514" 参数的值
$_114 = $_GET['114'];
$_514 = $_POST['514'];

// 检查 "114" 参数是否设置，并且其整数值大于 114514，
// 以及 "114" 的长度是否小于等于 3
if (isset($_114) && intval($_114) > 114514 && strlen($_114) <= 3) {
    // 检查 "514" 是否不是数字且大于 9999999
    if (!is_numeric($_514) && $_514 > 9999999) {
        $checker_3 = TRUE;
    }
}

// 从 $_POST 超全局变量中获取 "arr4y" 参数的值
$arr4y = $_POST['arr4y'];

// 检查 "arr4y" 是否为数组
if (is_array($arr4y)) {
    for ($i = 0; $i < count($arr4y); $i++) {
        // 检查 "arr4y" 的任何元素是否等于 "NSS"
        if ($arr4y[$i] === "NSS") {
            die("no!");
        }
        // 将 "arr4y" 的每个元素转换为整数
        $arr4y[$i] = intval($arr4y[$i]);
    }
    
    // 检查 "NSS" 是否在 "arr4y" 的第一个索引处
    if (array_search("NSS", $arr4y) === 0) {
        $checker_4 = TRUE;
    }
}

// 如果四个检查条件 ($checker_1、$checker_2、$checker_3、$checker_4) 都满足，则输出 $rce 的值
if ($checker_1 && $checker_2 && $checker_3 && $checker_4) {
    echo $rce;
}
?>
