vue学习

基础概念

是一套用于构建用户界面的渐进式框架。与其它大型框架不同的是，Vue 被设计为可以自底向上逐层应用。Vue 的核心库只关注视图层，不仅易于上手，还便于与第三方库或既有项目整合。另一方面，当与现代化的工具链以及各种支持类库结合使用时，Vue 也完全能够为复杂的单页应用提供驱动。

基础用法

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<div id="aa">
    {{massage}}
</div>

<script src="https://cdn.jsdelivr.net/npm/vue@2.5.21/dist/vue.min.js"></script>
<script>
    var vm = new Vue({
        el:"#aa",
        data:{
            massage:"小猫吃鱼"
        }
    });
</script>
</body>
</html>

if和for：

if:
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<div id="aa">
    <h1 v-if="massage==='a'">a</h1>

    <h1 v-else-if="massage==='c'">c</h1>

</div>

for:

<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<div id="aa">
   <li v-for="item in items">
       {{item.massage}}
   </li>
</div>

    var vm = new Vue({
        el:"#aa",

            items:[

</html>

axios异步通信：

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<div id="aa">
    <div>{{info.name}}</div>
    <div>{{info.age}}</div>
</div>

<script src="https://cdn.jsdelivr.net/npm/vue@2.5.21/dist/vue.min.js"></script>
<script src="https://unpkg.com/axios@1.6.5/dist/axios.min.js"></script>
<script type="text/javascript">
    var vm = new Vue({
        el:"#aa",
        data(){
            return{
                info: {
                    name:null,
                    age:null,
                    address:null
                }
            }
        },
        mounted(){
            axios.get('Vue/information.json').them(response=>(this.info=response.data))
        }
    });

</script>
</body>
</html>

    <title>Title</title>

<body>
<div id="aa">
    <p>AAA{{getTime()}}</p>

插槽slot

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Title</title>
</head>
<body>
<div id="aa">
  <topo>
      <topo-title slot="topo-title" :title="title"></topo-title>
      <topo-items slot="topo-items" v-for="item in topoitems" :item="item"></topo-items>
  </topo>
</div>

<script src="https://cdn.jsdelivr.net/npm/vue@2.5.21/dist/vue.min.js"></script>
<script>
  Vue.component("topo",{
    template: '<div>\
                <slot name="topo-title"></slot>\
                <ul>\
                <slot name="topo-items"></slot>\
                </ul>\
                </div>'
  });
  Vue.component("topo-title",{
    props:['title'],
    template: '<div>{{title}}</div>'
  });
  Vue.component("topo-items",{
    props:['item'],
    template:'<li>{{item}}</li>'
  });
  var vm = new Vue({
    el:"#aa",
    data:{
      title:"小猫吃鱼",
      topoitems:["鲈鱼","鲤鱼","鲫鱼"]
    }
  });
</script>
</body>
</html>
