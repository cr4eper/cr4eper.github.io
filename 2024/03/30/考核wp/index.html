要求密码在MD5加密后前六位为fcb4e6,gpt写脚本得字符串：

import hashlib
import random
import string

# 目标前六位MD5值
target_prefix = 'fcb4e6'

# 待加密字符串的长度
string_length = 6


# 定义一个用于生成随机字符串的函数
def generate_random_string(length):
    return ''.join(random.choices(string.ascii_lowercase + string.digits, k=length))


# 不断生成随机字符串并计算MD5哈希值，直到找到匹配的前六位
while True:
    # 生成随机字符串
    random_string = generate_random_string(string_length)

    # 计算MD5哈希值
    hash_value = hashlib.md5(random_string.encode()).hexdigest()

    # 检查前六位是否匹配目标前缀
    if hash_value[:6] == target_prefix:
        print("Found matching hash:", hash_value)
        print("Original string:", random_string)
        break
