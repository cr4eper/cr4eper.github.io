const express = require('express');  // 导入 Express 模块
const fs = require('fs');  // 导入文件系统模块
var bodyParser = require('body-parser');  // 导入用于解析请求体的中间件
const app = express();  // 创建 Express 应用程序实例

app.use(bodyParser.urlencoded({  // 使用 body-parser 中间件来解析 URL 编码的请求体
    extended: true
}));
app.use(bodyParser.json());  // 使用 body-parser 中间件来解析 JSON 格式的请求体

app.post('/plz', (req, res) => {  // 处理 POST /plz 路由的请求
    venom = req.body.venom;  // 从请求的请求体中获取名为 "venom" 的数据

    if (Object.keys(venom).length < 3 && venom.welcome == 159753) {  // 判断 "venom" 对象的属性数量和 "welcome" 属性的值
        try {
            if (venom.hasOwnProperty("text")) {  // 判断 "venom" 对象是否有属性 "text"
                res.send(venom.text);  // 如果有，则将其作为响应发送回客户端
            } else {
                res.send("no text detected");  // 如果没有，则发送 "no text detected" 作为响应
            }
        } catch {
            if (venom.text == "flag") {  // 判断 "venom" 对象的 "text" 属性值是否为 "flag"
                let flag = fs.readFileSync("C:\\Users\\35962\\Desktop\\web_hackjs_c9693952977bfa888e6f2d5f026b0692a8e4be97\\flag");  // 读取 "/flag" 文件的内容
                res.send("Congratulations:" + flag);  // 将读取到的内容作为响应发送回客户端
            } else {
                res.end("Nothing here!")  // 如果 "venom" 对象的 "text" 属性值不是 "flag"，则发送 "Nothing here!" 作为响应
            }
        }
    } else {
        res.end("happy game");  // 如果条件不满足，则发送 "happy game" 作为响应
    }
});

app.get('/', function (req, res, next) {  // 处理 GET / 路由的请求
    res.send('<title>oldjs</title><a>Hack me plz</a><br><form action="/plz" method="POST">text:<input type="text" name="venom[text]" value="ezjs"><input type="submit" value="Hack"></form>  ');
    // 将一个包含表单的 HTML 作为响应发送回客户端
});

app.listen(80, () => {  // 在端口 80 启动应用程序的监听
    console.log(`listening at port 80`);
});

要得到flag，我们就要构造请求体中venom的值，使其满足venom的数组长度小于三，且venom[welcome]==159735、venom[text]==flag；然后还要让其报错，才能捕获并执行下面获取flag的命令；

n8tz/CVE-2022-24999：“qs”原型中毒漏洞 （ CVE-2022-24999 ） (github.com)

大致含义就是我们可以在限定长度值时，如venom的长度<2时，我们将传入的参数设置为venom[__proto__][flag]=aaa&venom[text]=111时他依旧会判断为长度为小于二，但参数依旧存在

venom.hasOwnProperty(prop)： 方法会返回一个布尔值，指示对象自身属性中是否具有指定的属性。

参数：要检测的属性字符串名称或者

所以我们要令try字段报错的话，则修改hasOwnproperty()参数的值为数字即可。

所以我们可以有以下答案

venom[__proto__][text]=flag&venom[welcome]=159753&venom[hasOwnProperty]=1
