java–IO流

创建文件的三种方式

跟据路径创建文件

方法new File(String pathname)

package IOStream;
import java.io.File;
import java.io.IOException;

// 根据路径创建一个 File 对象
public class JavaIO {
    public static void main(String[] args) {
        createFile();
    }
    public static void createFile(){
        File file = new File("F:\\代码储存\\javaDeserialization\\untitled\\src\\IOStream\\new1.txt");
        try{
            file.createNewFile();
            System.out.println("Create Successfully");
        } catch (IOException e){
            e.printStackTrace();
        }
    }

}

此种会在当前代码文件夹之下生成一个文档

获取文件信息

package IOStream;

import java.io.File;

public class JavaIO3 {
    public static void main(String[] args) {
        getFileContents();
    }

    public static void getFileContents(){
        File file = new File("F:\\代码储存\\javaDeserialization\\untitled\\src\\IOStream\\new1.txt");
        System.out.println("文件名称为：" + file.getName());
        System.out.println("文件的绝对路径为：" + file.getAbsolutePath());
        System.out.println("文件的父级目录为：" + file.getParent());
        System.out.println("文件的大小(字节)为：" + file.length());
        System.out.println("这是不是一个文件：" + file.isFile());
        System.out.println("这是不是一个目录：" + file.isDirectory());
    }
}
