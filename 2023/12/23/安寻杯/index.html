def find_string():
    number = 0
    while True:
        number += 1
        string = str(number)
        first_hash = hashlib.md5(string.encode('utf-8')).hexdigest()
        second_hash = hashlib.md5(first_hash.encode('utf-8')).hexdigest()

def update(src, dst):
    if hasattr(dst, '__getitem__'):
        for key in src:
            if isinstance(src[key], dict):
                if key in dst and isinstance(src[key], dict):
                    update(src[key], dst[key])
                else:
                    dst[key] = src[key]
            else:
                dst[key] = src[key]
    else:
        for key, value in src.items():
            if hasattr(dst, key) and isinstance(value, dict):
                update(value, getattr(dst, key))
            else:
                setattr(dst, key, value)
